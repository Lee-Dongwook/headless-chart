  App.js: |
    import Widget from "@meursyphus/flitter-react";
    import { PieChart } from "@meursyphus/headless-chart";
    import {Text, Container, EdgeInsets, Row, Stack, Positioned, SizedBox, TextStyle} from '@meursyphus/flitter';

    const data = {
        labels: ["Red", "Blue", "Green", "Yellow"],
        datasets: [
            { legend: "#FF0000", values: [40] },
            { legend: "#0000FF", values: [30] },
            { legend: "#00FF00", values: [20] },
            { legend: "#FFFF00", values: [10] },
        ],
    };

    const chart = PieChart({
        data,
        getScale: () => ({
            total: 100,
        }),
        custom: {
            layout: (...[{legends, plot, title}]) => 
                Container({
                    padding: EdgeInsets.all(20),
                    child: Stack({
                        children: [
                            Positioned({
                                top: 0,
                                left: 0,
                                child: Text("Pie Chart Example", {
                                    style: new TextStyle({
                                        fontSize: 16,
                                        color: "#333333",
                                        fontWeight: "bold",
                                    }),
                                }),
                            }),
                            Row({
                                children: [
                                    SizedBox({width: 20}),
                                    plot,
                                ],
                            }),
                        ],
                    }),
                }),
                segment: ({arcs}) => Segments({arcs}),
                grid: ({radialLines, concentricCircles}) => 
                    new Grid({radialLines, concentricCircles}),
                legend: (...[{name}]) => new Legend({name}),
                title: (...[{name}]) => new Title({name}),
        },
    });

    export default function App() {
        return <Widget width="400px" height="400px" widget={chart} renderer="svg"/>;
    }

# Chartjs Inspired

A very simple headless-chart example inspired by Chart.js. All customizations are done inline for simplicity.
